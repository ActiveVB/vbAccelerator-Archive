' Helper classes to make combo boxes flat:
Private m_cboStyle As cFlatComboBox
Private m_cboSize As cFlatComboBox
' Helper class to perform drop-down tool windows:
Private m_cDW As cDropDownToolWindow

Private Sub HideDropDown()
   m_cDW.Hide
   tbrMain.ButtonChecked(15) = False
End Sub

Private Sub ShowTablePicker(ByVal x As Long, ByVal y As Long)
   ' Show the drop down:
   DefaultSize picDropDown
   m_cDW.Show x, y
   Draw m_cDW, 0, 0, 0

End Sub

Private Sub pResize(ByVal lRebarHeight As Long)
Dim lT As Long

   lT = lRebarHeight * Screen.TwipsPerPixelY
   On Error Resume Next
   txtDemo.Move 2 * Screen.TwipsPerPixelX, lT + 4 * Screen.TwipsPerPixelY,
    Me.ScaleWidth - 4 * Screen.TwipsPerPixelX, Me.ScaleHeight - lT - 6 *
    Screen.TwipsPerPixelY
      
End Sub

Private Sub Form_Load()
Dim i As Long
Dim sFont As String
Dim eStyle As ECTBToolButtonSyle

   With tbrMain
      .CreateToolbar 16
      .SetImageList ilsIconsMain, CTBImageListNormal
      For i = 1 To ilsIconsMain.ImageCount
         If (i = 10) Or (i = 11) Then
            eStyle = CTBDropDown
         ElseIf (i = 12) Then
            eStyle = CTBCheck
         Else
            eStyle = CTBNormal
         End If
         .AddButton , 14 - i, , , , eStyle
         If (i = 3) Or (i = 6) Or (i = 9) Or (i = 11) Or (i = 12) Then
            .AddButton , , , , , CTBSeparator
         End If
      Next i
   End With
   With tbrFormat
      .CreateToolbar 16, False, False, False, 16
      .SetImageList ilsIcons16, CTBImageListNormal
      For i = 1 To ilsIcons16.ImageCount
         .AddButton , 9 - i
         If (i = 3) Or (i = 7) Then
            .AddButton , , , , , CTBSeparator
         End If
      Next i
   End With
   With rbrMain
      .CreateRebar Me.hwnd
      .AddBandByHwnd tbrMain.hwnd, , , , "ToolbarBand"
      .AddBandByHwnd cboStyle.hwnd, , , , "ComboStyleBand"
      .AddBandByHwnd cboSize.hwnd, , False, , "ComboSizeBand"
      .AddBandByHwnd tbrFormat.hwnd, , False, , "FormatBand"
   End With
   txtDemo.Text = "This sample demonstrates using the Flat Combo Box class to
    give a flat appearance to two standard combo boxes added to a rebar."
   
   Set m_cboStyle = New cFlatComboBox
   m_cboStyle.Attach cboStyle
   Set m_cboSize = New cFlatComboBox
   m_cboSize.Attach cboSize
   
   ' Note this isn't the best way to get fonts.
   ' Use my font enumeration demo code to do it
   ' at least 10x quicker...
   For i = 0 To Printer.FontCount
      sFont = Printer.Fonts(i)
      cboStyle.AddItem sFont
      If (sFont = txtDemo.Font.Name) Then
         cboStyle.ListIndex = cboStyle.NewIndex
      End If
   Next i
   For i = 8 To 28
      cboSize.AddItem CStr(i)
   Next i
   cboSize.ListIndex = 0
   
   ' Initialise drop down helper class:
   Set m_cDW = New cDropDownToolWindow
   ' Set the object to show in the drop down:
   m_cDW.Create picDropDown
   
   
End Sub

Private Sub picDropDown_MouseDown(button As Integer, Shift As Integer, x As
 Single, y As Single)
   ' Mouse down handling:
   If (m_cDW.IsShown) Then
      ' Drop down window is visible
      If Not (m_cDW.InRect(x, y)) Then
         ' Mouse down outside drop-down area:
         HideDropDown
      Else
         ' Mouse down inside the drop down:
         Draw m_cDW, x \ Screen.TwipsPerPixelX, y \ Screen.TwipsPerPixelY,
          button
      End If
   End If
End Sub

Private Sub picDropDown_MouseMove(button As Integer, Shift As Integer, x As
 Single, y As Single)
   ' Mouse move.  Note that because all mouse messages are captured,
   ' x and y may be outside the limits of picDropdown.  This is
   ' handled in the draw routine.
   Draw m_cDW, x \ Screen.TwipsPerPixelX, y \ Screen.TwipsPerPixelY, button
End Sub

Private Sub picDropDown_MouseUp(button As Integer, Shift As Integer, x As
 Single, y As Single)
Dim xCellHit As Long, yCellHit As Long, bIn As Boolean

   ' Mouse up.  Determine whether mouse up over a cell:
   Draw m_cDW, x \ Screen.TwipsPerPixelX, y \ Screen.TwipsPerPixelY, button,
    bIn, xCellHit, yCellHit
   ' Hide the drop down:
   HideDropDown
   ' If an item selected, say what it was:
   If (bIn) Then
      MsgBox "Selected table: " & xCellHit & " x " & yCellHit, vbInformation
   End If
   
End Sub


Private Sub rbrMain_HeightChanged(lNewHeight As Long)
   pResize lNewHeight
End Sub

Private Sub tbrMain_ButtonClick(ByVal lButton As Long)
Dim x As Long, y As Long
   Debug.Print "ButtonClick:", lButton
   If (lButton = 15) Then
      tbrMain.GetDropDownPosition lButton, x, y
      ShowTablePicker x, y
   End If
End Sub
