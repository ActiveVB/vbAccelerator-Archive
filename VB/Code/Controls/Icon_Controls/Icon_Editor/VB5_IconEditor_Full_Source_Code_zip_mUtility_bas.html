 As Long, ByVal HPALETTE As Long, pccolorref As Long) As Long
Private Const CLR_INVALID = -1

Public Function TranslateColor(ByVal oClr As OLE_COLOR, _
                        Optional hPal As Long = 0) As Long
    ' Convert Automation color to Windows color
    If OleTranslateColor(oClr, hPal, TranslateColor) Then
        TranslateColor = CLR_INVALID
    End If
End Function

Public Sub gErr(ByVal lErrNum As Long)
Dim sMsg As String
Dim lErr As Long
   Select Case lErrNum
   Case 1
      sMsg = "Zoom must be in the range 1-32."
   Case 2
      sMsg = "Insufficient resources to continue."
   Case 3
      sMsg = "Error building memory device context."
   Case 4
      sMsg = "A Device Image with the specified size and number of colours
       already exists."
   Case 5
      sMsg = "Specified Device Image Index does not exist."
   Case 6
      sMsg = "Icon Width and Height must be greater than 0."
   Case 7
      sMsg = "Invalid Number of Icon Colours.  Please choose 2,16 or 256."
   Case 8
      sMsg = "Brush/Eraser/Line Width must be greater than 0 and less than 256."
   Case 9
      sMsg = "Unable to open the clipboard"
   Case 10
      sMsg = "Unable to load Icon"
   Case 11
      sMsg = "Device Images > 256 colours do not have a palette"
   End Select
   lErr = lErrNum + vbObjectError + 1048 + 27525
   Err.Raise lErr, App.EXEName & ".vbalIconEditor", sMsg
End Sub
