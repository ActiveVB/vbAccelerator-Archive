Private Declare Function IsWindow Lib "user32" (ByVal hWnd As Long) As Long
Private Declare Sub CopyMemory Lib "kernel32" Alias "RtlMoveMemory" (pDest As
 Any, pSrc As Any, ByVal ByteLen As Long)

Private m_lOwner As Long
Private m_hWnd As Long

Friend Sub Init(ByVal lPtr As Long, ByVal lhWnd As Long)
   m_lOwner = lPtr
   m_hWnd = lhWnd
End Sub

Private Property Get OwnerControl() As vbalToolBoxBarCtl
   If Not (m_hWnd = 0) Then
      If (IsWindow(m_hWnd)) Then
         If Not (m_lOwner = 0) Then
            Dim oT As Object
            CopyMemory oT, m_lOwner, 4
            Set OwnerControl = oT
            CopyMemory oT, 0&, 4
         End If
      End If
   End If
End Property

Public Property Get Count() As Long
Attribute Count.VB_Description = "Gets the number of bars associated with the
 control."
Dim oT As vbalToolBoxBarCtl
   Set oT = OwnerControl()
   If Not oT Is Nothing Then
      Count = oT.fBarCount
   End If
End Property

Public Property Get Item(Key As Variant) As cToolBoxBar
Attribute Item.VB_Description = "Gets a specific bar within the control by
 index or key."
Attribute Item.VB_UserMemId = 0
Dim oT As vbalToolBoxBarCtl
   Set oT = OwnerControl()
   If Not oT Is Nothing Then
      Set Item = oT.fBarItem(Key)
   End If
End Property

Public Sub Remove(Key As Variant)
Attribute Remove.VB_Description = "Removes a specific bar from the control."
Dim oT As vbalToolBoxBarCtl
   Set oT = OwnerControl()
   If Not oT Is Nothing Then
      oT.fBarRemove Key
   End If
End Sub

Public Function Add( _
      Optional Key As Variant, _
      Optional KeyBefore As Variant, _
      Optional Caption As String, _
      Optional IconIndex As Long = -1 _
   ) As cToolBoxBar
Attribute Add.VB_Description = "Adds a new bar to the control, optionally
 setting the caption and its position."
Dim oT As vbalToolBoxBarCtl
   Set oT = OwnerControl()
   If Not oT Is Nothing Then
      Set Add = oT.fBarAdd(Key, KeyBefore, Caption, IconIndex)
   End If
End Function
