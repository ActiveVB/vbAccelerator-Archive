Attribute VB_Name = "frmAniCursor"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Option Explicit

Private cAni As New cAniCursor
Private cClr As New cAniCursor
Private cC As New cCustomCursor

Private Sub cmdAniScreenPointer_Click()
Static bS As Boolean
   bS = Not (bS)
   cC.WaitCursor = bS
   If bS Then
      Dim lT As Long
      lT = Timer()
      ' do long operation
      Do While Timer - lT < 1
      Loop
   End If
End Sub

Private Sub cmdClassLong_Click()
Static bS As Boolean
   bS = Not (bS)
   If bS Then cClr.Destroy
   cAni.SetAsDefaultCursor Me.hWnd, bS
End Sub

Private Sub cmdStep_Click()
   cAni.Step
   cAni.Draw Me.hDC, 16, 16, , , Me.BackColor
   lblInfo.Caption = "Frame " & cAni.Frame + 1 & " of " & cAni.FrameCount
End Sub


Private Sub cmdSubClass_Click()
   If cC.hCursor = cAni.hCursor Then
      cC.Destroy
   Else
      cC.Attach Me.hWnd
      cC.hCursor = cAni.hCursor
   End If
   cmdAniScreenPointer.Enabled = (cC.hCursor <> 0)
End Sub

Private Sub cmdLoadResource_Click()
On Error GoTo ErrorHandler
   ' Under NT we can load directly...
   cAni.LoadFromResource 2001
   cmdStep_Click
   Exit Sub
ErrorHandler:
   MsgBox "Load From Resource only available under NT.", vbInformation
   cAni.LoadFromFile App.Path & "\globe.ani"
   Exit Sub
End Sub

Private Sub Form_Load()
   cClr.LoadFromFile App.Path & "\3dgarro.cur", , True
   cClr.SetAsDefaultCursor Me.hWnd, True
   cAni.LoadFromFile App.Path & "\globe.ani"
   lblInfo.Caption = "Frame " & cAni.Frame + 1 & " of " & cAni.FrameCount
End Sub

Private Sub Form_Paint()
   cAni.Draw Me.hDC, 16, 16
End Sub
