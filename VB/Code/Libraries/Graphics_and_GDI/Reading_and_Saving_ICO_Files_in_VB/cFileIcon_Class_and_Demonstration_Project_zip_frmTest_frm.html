Private m_cI As cFileIcon
Private Declare Function GetSystemDirectory Lib "kernel32" Alias
 "GetSystemDirectoryA" (ByVal lpBuffer As String, ByVal nSize As Long) As Long
Private Const MAX_PATH = 260

Private Property Get SystemDir() As String
Dim sPath As String
Dim lCount As Long
Dim iPos As Long
   sPath = String$(MAX_PATH, 0)
   lCount = GetSystemDirectory(sPath, MAX_PATH)
   If (lCount > 0) Then
      iPos = InStr(sPath, vbNullChar)
      If (iPos > 0) Then
         sPath = Left$(sPath, iPos - 1)
      End If
      SystemDir = sPath
   End If
End Property


Private Sub pDisplayIcons()
Dim i As Long
   lstIcons.Clear
   If m_cI.ImageCount > 0 Then
      For i = 1 To m_cI.ImageCount
         lstIcons.AddItem m_cI.ImageWidth(i) & "x" & m_cI.ImageHeight(i) & ", "
          & m_cI.ImageColourCount(i) & " Colours"
      Next i
      lstIcons.ListIndex = 0
   End If

End Sub

Private Sub cmdAdd_Click()
Dim i As Long
   Dim fA As New frmAddIcon
   fA.Show vbModal, Me
   If Not (fA.Cancelled) Then
      m_cI.AddImage fA.IconWidth, fA.IconHeight, fA.IconColours
      pDisplayIcons
   End If
End Sub

Private Sub lstIcons_Click()
Dim lIndex As Long
   lIndex = lstIcons.ListIndex + 1
   If (lIndex > 0) Then
      Set picIcon = m_cI.IconPicture(picIcon.hdc, lIndex)
      picImage.Cls
      picMask.Cls
      m_cI.DrawIconImage picImage.hdc, lIndex, ecfiImage
      m_cI.DrawIconImage picMask.hdc, lIndex, ecfiMask
   End If
End Sub
