Private Sub Form_Resize()
   On Error Resume Next
   txtData.Move 2 * Screen.TwipsPerPixelX, 2 * Screen.TwipsPerPixelY,
    Me.ScaleWidth - 4 * Screen.TwipsPerPixelX, Me.ScaleHeight - 4 *
    Screen.TwipsPerPixelY
End Sub
