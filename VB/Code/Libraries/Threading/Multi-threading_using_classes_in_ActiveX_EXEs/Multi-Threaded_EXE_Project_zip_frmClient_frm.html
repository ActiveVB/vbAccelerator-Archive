Attribute VB_Name = "frmClient"
Attribute VB_GlobalNameSpace = False

Attribute VB_PredeclaredId = True

Option Explicit

Private m_bCancel As Boolean
Private WithEvents m_cAD As cAsync
Attribute m_cAD.VB_VarHelpID = -1

Private Sub cmdStart_Click()
   m_bCancel = False
   m_cAD.Start
   cmdStop.Enabled = True
   cmdStart.Enabled = False
   cmdNonAsync.Enabled = False
End Sub

Private Sub cmdStop_Click()
   cmdStart.Enabled = False
   cmdStop.Enabled = False
   cmdNonAsync.Enabled = False
   m_bCancel = True
End Sub

Private Sub cmdNonAsync_Click()
   m_cAD.StartNonAsync
End Sub

Private Sub Form_Load()
   Set m_cAD = New cAsync
   ' 2 s interval for the demo (interval is
   ' measured in blocks of Sleep 100)
   m_cAD.Interval = 20
End Sub

Private Sub m_cAD_Cancelled()
   MsgBox "Cancel", vbInformation
   cmdStart.Enabled = True
   cmdNonAsync.Enabled = True
   cmdStop.Enabled = False
End Sub

Private Sub m_cAD_Complete()
   MsgBox "Complete", vbInformation
   cmdStart.Enabled = True
   cmdNonAsync.Enabled = True
   cmdStop.Enabled = False
End Sub

Private Sub m_cAD_Status(ByVal i As Long, Cancel As Boolean)
   Cancel = m_bCancel
End Sub
