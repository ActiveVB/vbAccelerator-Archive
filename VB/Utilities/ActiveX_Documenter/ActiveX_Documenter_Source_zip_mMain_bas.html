Option Explicit

Public Sub Main()
Dim tIccex As CommonControlsEx
   On Error Resume Next
   With tIccex
       .dwSize = LenB(tIccex)
       .dwICC = ICC_BAR_CLASSES
   End With
   InitCommonControlsEx tIccex
   On Error GoTo 0
